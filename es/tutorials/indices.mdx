<WereSoCool language={
`{ f: 310, l: 1, g: 1/4, p: 0 }\n
overtones = {
  Overlay [
    {2/1, 0, 1, 0},
    {1/1, 2, 1, 1},
    {1/1, 0, 1, -1},
    {1/2, 0, 1, 0},
  ]
}\n
-- Hay un puñado de índices diferentes que puede
-- se utiliza para seleccionar elementos de una lista.
main = {
  overtones |
  -- Temperamento igual de doce tonos.
  ET(12) 
  @ [
    -- Ya sabes que puedes elegir elementos por índice.
    0, 3, 4, 6, 9, 
    -- "Slices" se pueden utilizar para seleccionar rangos de elementos.
    -- este segmento seleccionará el 3er -> 7mo elemento de la lista
    -- El rango es inclusivo.
    3:7,
    -- "Slices" también puede retroceder. Nuevamente, las rebanadas son
    -- inclusive, por lo que elegirá los índices 8, 7, 6, 5, 4, 3 y 2.
    8:2,
    -- A "Slices" también se le puede pasar un tercer argumento llamado "skip".
    -- Esto comenzará desde el viaje del segundo elemento hasta el décimo.
    -- saltando 3 a la vez.
    0:11:3,
    -- Ya has visto Random(n). Que elige elementos aleatorios
    -- de la lista.
    Random(5),
    -- ¡También puede aplicar operaciones a cualquier índice!
    Random(30) | Overlay [Fm 1, Fm 9/8, Fm 15/8] | Gm 1/3,\n
    -- Silencio rápido. Agregaré un índice silencioso pronto. :)
    0 | Fm 0,\n
    -- ¡Por la victoria!
    11:2:2 
    | Fm 3/4 
    | Seq [Fm 1, Fm 9/8, Fm 4/3, Fm 5/4, Fm 3/2, Fm 5/3]
    | Overlay [Fm 1, Fm 3/2, Fm 2]
    | Gm 1/3,\n
    8:4 
    | Fm 1/2 
    | Lm 1/2,\n
    3 
    | Fm 1/2 
    | Lm 4
    | Overlay [Fm 1, Fm 9/4, Fm 3/2, Fm 5/4] | Gm 1/4
  ]
  | Lm 1/11
}`
}/>
